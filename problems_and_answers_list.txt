94. 問題:
   f'(t)+a\,f(t)=0,\quad f(0)=f_0

   答え:
   f(t)=f_0 e^{-a t}

--------------------------------------------------------------------------------

95. 問題:
   f'(t)+a\,f(t)=b

   答え:
   f(t)=\dfrac{b}{a}+\Big(f_0-\dfrac{b}{a}\Big)e^{-a t}

--------------------------------------------------------------------------------

96. 問題:
   f''(t)+a\,f'(t)=b

   答え:
   f(t)=f_0 +\dfrac{v_0 - \dfrac{b}{a}}{a} + \dfrac{b}{a}t-\dfrac{v_0-\dfrac{b}{a}}{a}e^{-a t}

--------------------------------------------------------------------------------

97. 問題:
   f'(t)+a\,f(t)=F\cos(\omega t)

   答え:
   f(t)=\dfrac{F}{\sqrt{a^2+\omega^2}}\cos(\omega t-\alpha)+\left(f_0-\dfrac{F a}{a^2+\omega^2}\right)e^{-a t},\quad \alpha=\arctan\left(\dfrac{\omega}{a}\right)

--------------------------------------------------------------------------------

98. 問題:
   \displaystyle \frac{\int_0^t f(s) \, ds}{a}+bf(t)=F\cos(\omega t),\quad f(0)=f_0

   答え:
   f(t)=\displaystyle \frac{F}{\sqrt{b^2+\left(\displaystyle \frac{1}{\omega a}\right)^2}}\cos(\omega t + \alpha)+\left(f_0+\displaystyle \frac{F b}{b^2+\left(\displaystyle \frac{1}{\omega a}\right)^2}\right)e^{- \frac{t}{ab}}

--------------------------------------------------------------------------------

99. 問題:
   Lf'(t)=A\sin(\omega t),\quad f(0)=0

   答え:
   f(t)=-\displaystyle \frac{A}{L\omega}\cos(\omega t)+\displaystyle \frac{A}{L\omega}

--------------------------------------------------------------------------------

100. 問題:

   答え:
   a_n = \displaystyle \frac{3n + 1}{n+1}

279. 問題:
   a_{n+2} + 3 a_{n+1} - 4 a_n = 0, \quad a_1 = 1,\ a_2 = 2

   答え:

280. 問題:
   a_{n+2} - a_{n+1} - 6 a_n = 0, \quad a_1 = 1,\ a_2 = 2

281. 問題:
   a_{n+2} - 4a_{n+1} + 4a_n = 0, \quad a_1 = 1,\ a_2 = 6

   答え:
   a_n = (-1 + 2n)2^{\,n-1}

282. 問題:
   a_{n+1} = 4a_n + b_n,\\[4pt]
b_{n+1} = -2a_n + b_n,\\[4pt]
a_1 = 2,\ b_1 = 1 \qquad (n\ge1)

   答え:
   \begin{cases}
  a_n = -3\cdot 2^{\,n-1} + 5\cdot 3^{\,n-1}\\[5pt]
  b_n = 6\cdot 2^{\,n-1} - 5\cdot 3^{\,n-1}\end{cases}

b_{n+1} = -a_n + 4b_n,\\[4pt]

   答え:
   \begin{cases}

284. 問題:
